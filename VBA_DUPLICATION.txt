Sub DUPLICATION()
    With Sheets("NOM")
        dl = .Cells(Rows.Count, 1).End(xlUp).Row
        Set ws = Sheets("DUPLI")
        k = 1
        For i = 1 To dl
            .Cells(i, 1).Copy ws.Cells(k, 1).Resize(6)
            k = k +
        Next i
    End With
End Sub

Private Sub Worksheet_BeforeDoubleClick(ByVal Target As Range, Cancel As Boolean)
